    template<typename T>
    struct H5T_COMPLEX_STRUCT {
        T real, imag; // real,imag parts
        using value_type     = T;
        using Scalar         = T;
        H5T_COMPLEX_STRUCT() = default;
        explicit H5T_COMPLEX_STRUCT(const std::complex<T> &in) {
            real = in.real();
            imag = in.imag();
        }
        H5T_COMPLEX_STRUCT &operator=(const std::complex<T> &rhs) {
            real = rhs.real();
            imag = rhs.imag();
            return *this;
        }
    };

    template<typename T>
    struct H5T_SCALAR2 {
